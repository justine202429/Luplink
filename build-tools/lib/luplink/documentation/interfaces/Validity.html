import { Component, Input, OnInit } from '@angular/core';
import { faCheck } from '@fortawesome/free-solid-svg-icons';

interface Validity {
  steps: { [key:string]: string },
  forms: { [key:string]: string },
}

@Component({
  selector: 'lib-stepbar',
  templateUrl: './stepbar.component.html',
  styleUrls: ['./stepbar.component.scss'],
})

export class StepbarComponent implements OnInit {
  @Input()
  get validity(): any {
    return this._validity;
  }
  set validity(value: any) {
    if (value) {
      console.log(value);
      this._validity = value;
    }
  }

  @Input()
  get step(): string {
    return this._step;
  }
  set step(value: string) {
    this._step = value;
  }
  private _validity: any; // Validity = { steps: {}, forms: {} };
  faCheck = faCheck;
  private _step: string = 'geometry';
  constructor() {
  }

  public isActive(step: string): boolean {
    // eslint-disable-next-line no-restricted-syntax
    for (const key in this.validity.steps) {
      if (Object.prototype.hasOwnProperty.call(this.validity.steps, key)) {
        // console.log(this.validity.steps);
        // console.log(key);
        if (this.validity.steps[key] === 'invalid' || this.step === step) {
          // We don't want the progress bar go beyond a warning
          return true;
        }
      }
    }
    return false;
  }
  ngOnInit(): void {
  }
}
