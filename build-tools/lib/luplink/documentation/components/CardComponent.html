import { Component, Input } from '@angular/core';
import { faAngleDown } from '@fortawesome/free-solid-svg-icons';

@Component({
  selector: 'lib-card',
  templateUrl: './card.component.html',
  styleUrls: ['./card.component.scss'],
})
export class CardComponent {
  @Input() public styleClass: string = '';

  @Input() title?: string;

  @Input() public isWide?: boolean = false;

  faAngleDown = faAngleDown;

  public collapsed: boolean = false;

  collapseCard() {
    this.collapsed = !this.collapsed;
  }
}

<div class="card" [ngClass]="styleClass" >
  <header class="card-header" *ngIf="title">
    <p class="card-header-title">{{title}}</p>
    <a class="card-header-icon is-text is-link" (click)="collapseCard()" aria-label="collapse">
      <span class="icon"><fa-icon [icon]="faAngleDown" transform="rotate-{{collapsed ? 0 : 180}}" aria-hidden="true"></fa-icon></span>
    </a>
  </header>

  <div class="card-content" *ngIf="!collapsed">
    <div class="content" >
      <ng-content></ng-content>
    </div>
  </div>
</div>
